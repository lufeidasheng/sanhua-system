# 音频配置模型 (企业级增强)
class AudioConfig(BaseModel):
    rate: int = Field(default=16000, ge=8000, le=48000, description="采样率")
    chunk: int = Field(default=1024, ge=256, le=4096, description="块大小")
    channels: int = Field(default=1, ge=1, le=2, description="通道数")
    format: int = Field(default=pyaudio.paInt16, description="音频格式")
    device_index: Optional[int] = Field(default=None, description="设备索引")
    silence_threshold: int = Field(default=500, ge=50, le=2000, description="静音阈值")
    max_silence_duration: float = Field(default=1.0, gt=0.1, le=10.0, description="最大静音时长")
    max_recording_duration: float = Field(default=30.0, gt=1.0, le=600.0, description="最大录音时长")
    noise_reduction_level: float = Field(default=0.7, ge=0.1, le=1.0, description="降噪级别")
    normalization: bool = Field(default=True, description="是否启用归一化")
    output_dir: str = Field(default="/var/lib/voice_module/recordings", description="录音存储目录")

    @validator('format', pre=True)
    def validate_format(cls, v):
        if v not in [pyaudio.paInt16, pyaudio.paFloat32]:
            raise ValueError("不支持的音频格式，仅支持 paInt16 或 paFloat32")
        return v
    
    @validator('output_dir', pre=True)
    def validate_output_dir(cls, v):
        if not os.path.exists(v):
            os.makedirs(v, exist_ok=True, mode=0o755)
        return v

@dataclass
class AudioState:
    is_recording: bool = False
    is_paused: bool = False
    duration: float = 0.0
    frames_count: int = 0
    last_audio_time: float = 0.0
    error: Optional[str] = None
    cpu_usage: float = 0.0
    memory_usage: int = 0

class VoiceInput:
    def __init__(self, config: Optional[Dict[str, Any]] = None):
        # 加载企业级配置
        self.config = self._load_enterprise_config(config)
        
        # 初始化 PyAudio
        self._pyaudio = pyaudio.PyAudio()
        self._stream = None
        
        # 状态管理
        self.state = AudioState()
        self._recording_event = threading.Event()
        self._recording_paused = threading.Event()
        self._recording_paused.clear()
        
        # 数据队列
        self._sync_audio_queue = queue.Queue(maxsize=100)  # 限制队列大小防止内存溢出
        self._frames: List[bytes] = []
        
        # 时间追踪
        self._recording_start_time: Optional[float] = None
        
        # 线程和任务
        self._recording_thread: Optional[threading.Thread] = None
        self._audio_task: Optional[asyncio.Task] = None
        self._loop: Optional[asyncio.AbstractEventLoop] = None
        
        # 异常处理
        self._exception: Optional[Exception] = None
        self._active = True
        
        # 事件发布
        self._event_bus = event_bus
        
        # 性能监控
        self._monitor_task: Optional[asyncio.Task] = None
        
        logger.info("语音输入实例已创建")

    def _load_enterprise_config(self, config: Optional[Dict[str, Any]]) -> AudioConfig:
        """加载企业级配置，合并默认配置、系统配置和传入配置"""
        # 加载系统默认配置
        final_config = DEFAULT_CONFIG.copy()
        
        # 加载企业配置文件
        if os.path.exists(CONFIG_PATH):
            try:
                with open(CONFIG_PATH, 'r') as f:
                    system_config = json.load(f)
                final_config.update(system_config)
                logger.info("加载企业配置文件: %s", CONFIG_PATH)
            except Exception as e:
                logger.error("加载配置文件失败: %s", str(e))
        
        # 合并传入配置
        if config:
            final_config.update(config)
        
        return AudioConfig(**final_config)

    async def __aenter__(self):
        """异步上下文管理器入口"""
        await self.start_recording()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """异步上下文管理器退出"""
        self.terminate()

    def _preprocess_audio(self, data: bytes) -> bytes:
        """
        企业级音频预处理:
        1. 自适应降噪
        2. 归一化
        3. 音频压缩
        """
        # 转换为numpy数组
        audio_np = np.frombuffer(data, dtype=np.int16)
        
        # 跳过空数据
        if audio_np.size == 0:
            return data
        
        # 自适应降噪处理
        rms = np.sqrt(np.mean(np.square(audio_np)))
        threshold = max(
            int(self.config.silence_threshold * self.config.noise_reduction_level),
            int(rms * 0.8)  # 自适应阈值
        )
        audio_np = np.where(np.abs(audio_np) < threshold, 0, audio_np)
        
        # 归一化处理
        if self.config.normalization:
            max_val = np.max(np.abs(audio_np))
            if max_val > 0:
                scale_factor = 32767 / max_val
                audio_np = (audio_np * scale_factor).astype(np.int16)
        
        return audio_np.tobytes()

    async def start_recording(self, duration: Optional[float] = None):
        if self._recording_event.is_set():
            logger.warning("录音已在进行中")
            return
        if not self._active:
            raise RuntimeError("实例已被终止")

        # 重置状态
        self._frames.clear()
        self._recording_event.set()
        self._recording_paused.clear()
        self._recording_start_time = time.time()
        self._exception = None
        
        # 更新状态
        self.state = AudioState(
            is_recording=True,
            is_paused=False,
            duration=0.0,
            frames_count=0
        )
        
        # 发布事件
        await self._event_bus.publish("recording_started", {"duration": duration})

        try:
            # 打开音频流
            self._stream = self._pyaudio.open(
                format=self.config.format,
                channels=self.config.channels,
                rate=self.config.rate,
                input=True,
                frames_per_buffer=self.config.chunk,
                input_device_index=self.config.device_index,
                stream_callback=self._stream_callback if sys.platform == 'linux' else None
            )
            logger.info("音频流已打开")
        except Exception as e:
            self._recording_event.clear()
            logger.error("打开音频流失败: %s", str(e))
            self.state.error = str(e)
            await self._event_bus.publish("recording_error", {"error": str(e)})
            raise

        # 获取事件循环
        try:
            self._loop = asyncio.get_running_loop()
        except RuntimeError:
            self._loop = asyncio.new_event_loop()
            asyncio.set_event_loop(self._loop)

        # 启动录音线程
        self._recording_thread = threading.Thread(
            target=self._record_audio,
            args=(duration,),
            daemon=True,
            name="VoiceInput-RecordingThread"
        )
        self._recording_thread.start()

        # 启动异步音频处理任务
        self._audio_task = self._loop.create_task(self._process_audio_queue())
        
        # 启动性能监控
        self._monitor_task = self._loop.create_task(self._monitor_performance())

        logger.info("录音已开始, 时长: %s", duration or '无限制')

    def _stream_callback(self, in_data, frame_count, time_info, status):
        """Linux系统专用回调函数，提高实时性"""
        try:
            if not self._recording_event.is_set() or self._recording_paused.is_set():
                return (in_data, pyaudio.paContinue)
            
            # 预处理音频
            processed_data = self._preprocess_audio(in_data)
            
            # 放入队列
            self._sync_audio_queue.put(processed_data, block=False)
            
            return (in_data, pyaudio.paContinue)
        except Exception as e:
            logger.error("音频流回调错误: %s", str(e))
            self._sync_audio_queue.put(e)
            return (in_data, pyaudio.paAbort)

    def pause_recording(self):
        if not self._recording_event.is_set():
            return
            
        self._recording_paused.set()
        self.state.is_paused = True
        logger.info("录音已暂停")
        asyncio.run_coroutine_threadsafe(
            self._event_bus.publish("recording_paused", {}),
            self._loop
        )

    def resume_recording(self):
        if not self._recording_event.is_set():
            return
            
        self._recording_paused.clear()
        self.state.is_paused = False
        logger.info("录音已恢复")
        asyncio.run_coroutine_threadsafe(
            self._event_bus.publish("recording_resumed", {}),
            self._loop
        )

    def stop_recording(self) -> float:
        if not self._recording_event.is_set():
            return 0.0

        self._recording_event.clear()
        recording_end_time = time.time()
        
        # 更新状态
        duration = self.get_audio_duration()
        self.state.is_recording = False
        self.state.duration = duration

        # 取消任务
        if self._audio_task:
            self._audio_task.cancel()
        if self._monitor_task:
            self._monitor_task.cancel()

        # 等待线程结束
        if self._recording_thread and self._recording_thread.is_alive():
            self._recording_thread.join(timeout=2.0)
            if self._recording_thread.is_alive():
                logger.warning("录音线程未能在超时内终止")

        # 发布事件
        asyncio.run_coroutine_threadsafe(
            self._event_bus.publish("recording_stopped", {
                "duration": duration,
                "frames_count": len(self._frames)
            }),
            self._loop
        )

        logger.info("录音已停止, 时长: %.2f秒", duration)
        return duration

    def terminate(self):
        if not self._active:
            return
            
        self.stop_recording()

        # 关闭音频流
        if self._stream:
            try:
                if self._stream.is_active():
                    self._stream.stop_stream()
                self._stream.close()
                logger.info("音频流已关闭")
            except Exception as e:
                logger.error("关闭音频流失败: %s", str(e))
            finally:
                self._stream = None

        # 终止PyAudio
        try:
            self._pyaudio.terminate()
            logger.info("PyAudio已终止")
        except Exception as e:
            logger.error("终止PyAudio失败: %s", str(e))

        self._active = False
        logger.info("资源已释放")

    def get_audio_duration(self) -> float:
        if not self._recording_start_time:
            return 0.0
            
        if self._recording_event.is_set():
            return time.time() - self._recording_start_time
        else:
            return self.state.duration

    def get_recorded_frames(self) -> List[bytes]:
        return self._frames.copy()

    def _record_audio(self, duration: Optional[float]):
        """录音线程主循环"""
        logger.debug("录音线程启动")
        start_time = time.time()
        
        try:
            while self._recording_event.is_set():
                if self._recording_paused.is_set():
                    time.sleep(0.05)
                    continue
                    
                try:
                    # 读取音频数据
                    data = self._stream.read(
                        self.config.chunk, 
                        exception_on_overflow=False
                    )
                    
                    # 预处理音频
                    processed_data = self._preprocess_audio(data)
                    
                    # 放入队列
                    self._sync_audio_queue.put(processed_data, timeout=0.5)
                    
                    # 检查时长限制
                    if duration and (time.time() - start_time) >= duration:
                        logger.info("达到录制时长 %d秒，停止录音", duration)
                        self._recording_event.clear()
                        
                except queue.Full:
                    logger.warning("音频队列已满，丢弃部分数据")
                except Exception as e:
                    self._exception = e
                    logger.exception("录音线程处理异常")
                    self._sync_audio_queue.put(e)
                    
                time.sleep(0.01)
                
        except Exception as e:
            self._exception = e
            logger.exception("录音线程异常")
            self._sync_audio_queue.put(e)
            
        finally:
            self._recording_event.clear()
            logger.debug("录音线程退出")

    async def _process_audio_queue(self):
        """异步音频处理任务"""
        logger.debug("音频处理任务启动")
        self.state.last_audio_time = time.time()
        
        try:
            while self._recording_event.is_set() or not self._sync_audio_queue.empty():
                try:
                    # 从同步队列获取数据
                    item = await asyncio.get_running_loop().run_in_executor(
                        None, 
                        lambda: self._sync_audio_queue.get(timeout=1)
                    )
                    
                    if isinstance(item, Exception):
                        logger.error("录音线程传递异常: %s", str(item))
                        self.state.error = str(item)
                        await self._event_bus.publish("recording_error", {"error": str(item)})
                        break
                        
                    # 添加到帧列表
                    self._frames.append(item)
                    self.state.frames_count = len(self._frames)
                    self.state.last_audio_time = time.time()
                    
                    # 定期发布状态
                    if len(self._frames) % 20 == 0:  # 每20帧发布一次
                        await self._event_bus.publish("audio_data", {
                            "frame_count": len(self._frames),
                            "duration": self.get_audio_duration()
                        })
                        
                except queue.Empty:
                    if not self._recording_event.is_set():
                        break
                        
        except asyncio.CancelledError:
            logger.info("音频处理任务被取消")
        except Exception as e:
            logger.error("音频处理任务异常: %s", str(e))
            self.state.error = str(e)
            await self._event_bus.publish("recording_error", {"error": str(e)})
        finally:
            logger.debug("音频处理任务退出")
            
    async def _monitor_performance(self):
        """性能监控任务"""
        try:
            import psutil
            process = psutil.Process(os.getpid())
            
            while self._recording_event.is_set():
                # CPU使用率
                cpu_percent = process.cpu_percent(interval=1)
                # 内存使用
                memory_info = process.memory_info()
                
                # 更新状态
                self.state.cpu_usage = cpu_percent
                self.state.memory_usage = memory_info.rss
                
                # 发布性能事件
                if cpu_percent > 80:  # 高CPU使用警告
                    await self._event_bus.publish("performance_warning", {
                        "cpu_usage": cpu_percent,
                        "memory_usage": memory_info.rss
                    })
                
                await asyncio.sleep(5)
                
        except ImportError:
            logger.warning("psutil 未安装，无法监控性能")
        except Exception as e:
            logger.error("性能监控失败: %s", str(e))

    def save_wav(self, filename: str):
        """保存录音到文件，符合企业存储规范"""
        if not self._frames:
            raise ValueError("没有可保存的音频数据")
            
        try:
            # 确保输出目录存在
            output_dir = self.config.output_dir
            os.makedirs(output_dir, exist_ok=True, mode=0o755)
            
            # 构建完整路径
            full_path = os.path.join(output_dir, filename)
            
            with wave.open(full_path, 'wb') as wf:
                wf.setnchannels(self.config.channels)
                wf.setsampwidth(self._pyaudio.get_sample_size(self.config.format))
                wf.setframerate(self.config.rate)
                wf.writeframes(b''.join(self._frames))
                
            # 设置适当权限
            os.chmod(full_path, 0o644)
            
            logger.info("音频已保存至: %s", full_path)
            asyncio.run_coroutine_threadsafe(
                self._event_bus.publish("audio_saved", {"filename": full_path}),
                self._loop
            )
            return full_path
        except Exception as e:
            logger.error("保存WAV文件失败: %s", str(e))
            self.state.error = str(e)
            asyncio.run_coroutine_threadsafe(
                self._event_bus.publish("save_error", {"error": str(e)}),
                self._loop
            )
            raise

    async def record_until_silence(
        self,
        silence_threshold: Optional[int] = None,
        silence_duration: Optional[float] = None,
        max_duration: Optional[float] = None
    ) -> float:
        """企业级静音检测录音"""
        threshold = silence_threshold or self.config.silence_threshold
        max_silence = silence_duration or self.config.max_silence_duration
        max_recording = max_duration or self.config.max_recording_duration

        await self.start_recording()

        start_time = time.time()
        silence_start = None
        last_audio_time = start_time

        logger.info(
            "开始静音检测录音: 阈值=%d, 最大静音=%.1fs, 最大录音=%.1fs",
            threshold, max_silence, max_recording
        )

        try:
            while self._recording_event.is_set():
                try:
                    # 异步获取音频数据
                    audio_data = await asyncio.wait_for(
                        asyncio.get_running_loop().run_in_executor(
                            None, 
                            lambda: self._sync_audio_queue.get(timeout=0.5)
                        ),
                        timeout=0.5
                    )
                    
                    if audio_data:
                        # 分析音频振幅
                        audio_np = np.frombuffer(audio_data, dtype=np.int16)
                        amplitude = np.max(np.abs(audio_np)) if audio_np.size > 0 else 0
                        last_audio_time = time.time()

                        # 检测静音
                        if amplitude > threshold:
                            silence_start = None
                        else:
                            if silence_start is None:
                                silence_start = time.time()
                            elif (time.time() - silence_start) >= max_silence:
                                logger.info("检测到持续静音，停止录音")
                                break
                except (asyncio.TimeoutError, queue.Empty):
                    # 处理超时或空队列
                    current_time = time.time()
                    
                    # 检查静音超时
                    if silence_start and (current_time - silence_start) >= max_silence:
                        logger.info("检测到持续静音(超时)，停止录音")
                        break
                        
                    # 检查无音频输入超时
                    if (current_time - last_audio_time) >= max_silence:
                        logger.info("长时间无音频输入(超时)，停止录音")
                        break
                        
                    # 检查最大录音时长
                    if (current_time - start_time) >= max_recording:
                        logger.info("达到最大录音时长(超时)，停止录音")
                        break
                        
        except Exception as e:
            logger.error("静音检测录音异常: %s", str(e))
            self.state.error = str(e)
            await self._event_bus.publish("recording_error", {"error": str(e)})

        duration = self.stop_recording()
        return duration


# ========== 企业级模块接口 ==========

voice_input_instance: Optional[VoiceInput] = None
module_active = False

async def init(config: Optional[Dict[str, Any]] = None) -> bool:
    """初始化语音输入模块"""
    global voice_input_instance, module_active
    logger.info("初始化语音输入模块")

    # 清理现有实例
    if voice_input_instance:
        voice_input_instance.terminate()

    voice_input_instance = None
    module_active = True
    
    # 发布初始化事件
    await event_bus.publish("module_initialized", {"module": "voice_input"})
    return True

async def start(config: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
    """启动语音输入模块"""
    global voice_input_instance, module_active

    if not module_active:
        return {"status": "error", "message": "模块未初始化"}

    logger.info("启动语音输入模块")

    try:
        voice_input_instance = VoiceInput(config=config)
        await event_bus.publish("module_started", {"module": "voice_input"})
        
        return {
            "status": "success",
            "config": voice_input_instance.config.dict()
        }
    except Exception as e:
        logger.error("启动失败: %s", str(e))
        await event_bus.publish("module_error", {
            "module": "voice_input",
            "error": str(e)
        })
        return {"status": "error", "message": str(e)}

async def stop() -> bool:
    """停止语音输入模块"""
    global voice_input_instance, module_active
    logger.info("停止语音输入模块")

    if voice_input_instance:
        voice_input_instance.terminate()
        voice_input_instance = None

    module_active = False
    await event_bus.publish("module_stopped", {"module": "voice_input"})
    return True

async def on_event(event_type: str, data: Dict[str, Any]) -> Dict[str, Any]:
    """处理企业级模块事件"""
    global voice_input_instance

    if not module_active:
        return {"status": "error", "message": "模块未激活"}

    if voice_input_instance is None:
        return {"status": "error", "message": "语音输入实例未创建"}

    try:
        # 启动录音
        if event_type == "start_recording":
            duration = data.get("duration")
            await voice_input_instance.start_recording(duration=duration)
            return {"status": "success", "action": "recording_started"}
        
        # 停止录音
        elif event_type == "stop_recording":
            duration = voice_input_instance.stop_recording()
            return {"status": "success", "action": "recording_stopped", "duration": duration}
        
        # 暂停录音
        elif event_type == "pause_recording":
            voice_input_instance.pause_recording()
            return {"status": "success", "action": "recording_paused"}
        
        # 恢复录音
        elif event_type == "resume_recording":
            voice_input_instance.resume_recording()
            return {"status": "success", "action": "recording_resumed"}
        
        # 保存录音
        elif event_type == "save_wav":
            filename = data.get("filename", "output.wav")
            saved_path = voice_input_instance.save_wav(filename)
            return {"status": "success", "action": "audio_saved", "path": saved_path}
        
        # 静音检测录音
        elif event_type == "record_until_silence":
            result = await voice_input_instance.record_until_silence(
                silence_threshold=data.get("silence_threshold"),
                silence_duration=data.get("silence_duration"),
                max_duration=data.get("max_duration")
            )
            return {
                "status": "success",
                "action": "recording_stopped_due_to_silence",
                "duration": result
            }
        
        # 获取状态
        elif event_type == "get_status":
            return {
                "status": "success",
                "state": {
                    "recording": voice_input_instance.state.is_recording,
                    "paused": voice_input_instance.state.is_paused,
                    "duration": voice_input_instance.state.duration,
                    "frames_count": voice_input_instance.state.frames_count,
                    "last_audio_time": voice_input_instance.state.last_audio_time,
                    "error": voice_input_instance.state.error,
                    "cpu_usage": voice_input_instance.state.cpu_usage,
                    "memory_usage": voice_input_instance.state.memory_usage
                }
            }
        
        # 更新配置
        elif event_type == "update_config":
            new_config = data.get("config", {})
            if voice_input_instance:
                try:
                    # 创建新配置（合并更新）
                    updated_config = voice_input_instance.config.copy(update=new_config)
                    voice_input_instance.config = updated_config
                    return {"status": "success", "config": updated_config.dict()}
                except Exception as e:
                    return {"status": "error", "message": str(e)}
            return {"status": "error", "message": "实例未创建"}
        
        return {"status": "error", "message": "未知事件类型"}

    except Exception as e:
        logger.error("处理事件 %s 时出错: %s", event_type, str(e))
        await event_bus.publish("module_error", {
            "module": "voice_input",
            "event": event_type,
            "error": str(e)
        })
        return {"status": "error", "message": str(e)}

    dispatcher.register_action(

    )

    dispatcher.register_action(

    return {"status": "success", "action": "recording_started"}

    if not voice_input_instance:
        return {"status": "error", "message": "语音输入模块未初始化"}
    
    duration = voice_input_instance.stop_recording()
    return {"status": "success", "duration": duration}

        silence_duration=kwargs.get("silence_duration"),

    )
    return {"status": "success", "duration": duration}

async def save_recording_action(context, filename: str = "output.wav"):

# ========== 模块入口点和测试函数 ==========

def entry():
    """企业级模块入口函数"""
    logger.info("语音输入模块入口点调用")
    
    # 创建事件循环
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)
    
    try:
        # 初始化并启动模块
        loop.run_until_complete(init())
        start_result = loop.run_until_complete(start())
        
        if start_result["status"] != "success":
            logger.error("模块启动失败: %s", start_result.get("message", "未知错误"))
            return False
            
        logger.info("语音输入模块已成功启动")
        
        # 保持运行（可根据需要调整）
        loop.run_forever()
        return True
    except KeyboardInterrupt:
        logger.info("模块被用户中断")
        return True
    except Exception as e:
        logger.exception("模块运行异常: %s", str(e))
        return False
    finally:
        # 清理资源
        loop.run_until_complete(stop())
        loop.close()
        logger.info("语音输入模块已关闭")
