# 初始化日志
def setup_logger(log_file: Path = None, log_level: str = "INFO"):
    """配置企业级日志系统"""
    logger = logging.getLogger("STTModule")
    logger.setLevel(log_level)
    
    formatter = logging.Formatter(
        "%(asctime)s - %(name)s - %(levelname)s - %(process)d - %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S"
    )
    
    # 控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(formatter)
    logger.addHandler(console_handler)
    
    # 文件处理器（如果指定了日志文件）
    if log_file:
        log_file.parent.mkdir(parents=True, exist_ok=True)
        file_handler = logging.FileHandler(log_file, encoding='utf-8')
        file_handler.setFormatter(formatter)
        logger.addHandler(file_handler)
    
    return logger

@dataclass
class STTConfig:
    """语音识别模块配置"""
    model_size: str = "base"  # Whisper模型大小 [tiny, base, small, medium, large]
    device: Optional[str] = None  # 指定设备 (cuda, cpu, cuda:0)
    language: str = "auto"  # 识别语言 (auto, zh, en, etc.)
    compute_type: str = "float16"  # 计算精度 (float32, float16, int8)
    beam_size: int = 5  # Beam搜索大小
    temperature: float = 0.0  # 采样温度
    log_level: str = "INFO"  # 日志级别
    log_file: Optional[str] = None  # 日志文件路径
    max_concurrent: int = 3  # 最大并发任务数
    max_audio_duration: int = 300  # 最大音频时长(秒)
    max_audio_size: int = 50 * 1024 * 1024  # 最大音频文件大小(50MB)
    temp_dir: str = "/var/tmp/stt_processing"  # 临时文件目录
    health_check_interval: int = 60  # 健康检查间隔(秒)
    auto_fallback: bool = True  # 是否启用自动降级
    fallback_model: str = "base"  # 降级模型
    gpu_threshold: float = 0.8  # GPU使用率阈值
    cpu_threshold: float = 0.7  # CPU使用率阈值
    disk_threshold: float = 0.9  # 磁盘使用率阈值

    @classmethod
    def from_dict(cls, config_dict: Dict) -> "STTConfig":
        """从字典创建配置对象"""
        return cls(**config_dict)

@dataclass
class TranscriptionTask:
    """语音转写任务"""
    audio_file: str
    task_id: str = ""
    callback_queue: Optional[mp.Queue] = None
    metadata: Dict = field(default_factory=dict)
    timestamp: float = field(default_factory=time.time)

class ResourceMonitor:
    """系统资源监控器"""
    def __init__(self, config: STTConfig):
        self.config = config
        self.logger = logging.getLogger("STTModule.ResourceMonitor")
        self.last_check = time.time()
        self.stats = {
            "gpu_usage": 0.0,
            "cpu_usage": 0.0,
            "mem_usage": 0.0,
            "disk_usage": 0.0,
            "free_disk": 0
        }
    
    def check_resources(self) -> bool:
        """检查系统资源是否可用"""
        current_time = time.time()
        # 按配置间隔检查，避免频繁检查
        if current_time - self.last_check < self.config.health_check_interval / 2:
            return True
        
        self.last_check = current_time
        try:
            # 获取CPU使用率
            self.stats["cpu_usage"] = psutil.cpu_percent(interval=0.1) / 100
            
            # 获取内存使用率
            mem = psutil.virtual_memory()
            self.stats["mem_usage"] = mem.percent / 100
            
            # 获取磁盘使用率
            disk = psutil.disk_usage('/')
            self.stats["disk_usage"] = disk.percent / 100
            self.stats["free_disk"] = disk.free
            
            # 获取GPU使用率（如果可用）
            if torch.cuda.is_available():
                self.stats["gpu_usage"] = get_gpu_utilization()
            
            # 检查是否超过阈值
            if (self.stats["cpu_usage"] > self.config.cpu_threshold or
                self.stats["gpu_usage"] > self.config.gpu_threshold or
                self.stats["disk_usage"] > self.config.disk_threshold):
                self.logger.warning(
                    f"资源使用过高: CPU={self.stats['cpu_usage']*100:.1f}%, "
                    f"GPU={self.stats['gpu_usage']*100:.1f}%, "
                    f"Disk={self.stats['disk_usage']*100:.1f}%"
                )
                return False
            return True
        except Exception as e:
            self.logger.error(f"资源监控失败: {e}")
            return True  # 监控失败时不阻塞处理

class STTModule:
    """企业级语音识别模块"""
    def __init__(self, command_queue: mp.Queue, result_queue: mp.Queue, 
                 stop_event: mp.Event, config: STTConfig):
        self.command_queue = command_queue
        self.result_queue = result_queue
        self.stop_event = stop_event
        self.config = config
        self.model = None
        self.task_counter = 0
        self.active_tasks = {}
        self.lock = mp.Lock()
        self.model_load_time = 0
        
        # 初始化日志
        self.logger = setup_logger(
            Path(self.config.log_file) if self.config.log_file else None,
            self.config.log_level
        )
        
        # 创建临时目录
        Path(self.config.temp_dir).mkdir(parents=True, exist_ok=True)
        
        # 初始化资源监控器
        self.resource_monitor = ResourceMonitor(config)
        
        self.logger.info(f"STT模块初始化完成，配置: {self.config}")
        self.load_model()

    def load_model(self):
        """加载语音识别模型"""
        if self.model is not None:
            return
        
        start_time = time.time()
        self.logger.info(f"正在加载Whisper模型（大小：{self.config.model_size}）...")
        
        try:
            # 确定运行设备
            device = self.config.device or ("cuda" if torch.cuda.is_available() else "cpu")
            
            # 加载模型
            self.model = whisper.load_model(
                self.config.model_size, 
                device=device,
                download_root="/var/cache/whisper/models"
            )
            
            # 设置模型选项
            options = whisper.DecodingOptions(
                language=self.config.language,
                beam_size=self.config.beam_size,
                fp16=(self.config.compute_type == "float16"),
                temperature=self.config.temperature
            )
            self.model.decoding_options = options
            
            self.model_load_time = time.time() - start_time
            self.logger.info(
                f"模型加载完成，耗时: {self.model_load_time:.2f}s, "
                f"设备: {device}, 参数: {self.model.decoding_options}"
            )
            
        except Exception as e:
            self.logger.error(f"模型加载失败: {e}", exc_info=True)
            # 尝试加载降级模型
            if self.config.auto_fallback and self.config.fallback_model != self.config.model_size:
                self.logger.warning(f"尝试加载降级模型: {self.config.fallback_model}")
                self.config.model_size = self.config.fallback_model
                self.load_model()
            else:
                self.logger.critical("模型加载失败且无法降级，模块无法运行")
                self.stop_event.set()

    def validate_audio_file(self, audio_path: str) -> bool:
        """验证音频文件是否符合要求"""
        try:
            # 检查文件是否存在
            if not os.path.isfile(audio_path):
                self.logger.error(f"音频文件不存在: {audio_path}")
                return False
            
            # 检查文件大小
            file_size = os.path.getsize(audio_path)
            if file_size == 0:
                self.logger.error(f"空音频文件: {audio_path}")
                return False
            if file_size > self.config.max_audio_size:
                self.logger.error(f"音频文件过大 ({file_size/1024/1024:.2f}MB > {self.config.max_audio_size/1024/1024:.2f}MB): {audio_path}")
                return False
            
            # 检查磁盘空间
            free_space = get_free_disk_space(audio_path)
            if free_space < 100 * 1024 * 1024:  # 100MB
                self.logger.warning(f"磁盘空间不足: {free_space/1024/1024:.2f}MB")
                return False
            
            return True
        except Exception as e:
            self.logger.error(f"音频文件验证失败: {e}")
            return False

    def transcribe_audio(self, task: TranscriptionTask) -> Dict:
        """执行语音转写任务"""
        result = {
            "task_id": task.task_id,
            "audio_file": task.audio_file,
            "timestamp": time.time(),
            "model": self.config.model_size
        }
        
        # 验证音频文件
        if not self.validate_audio_file(task.audio_file):
            result.update({"status": "error", "error": "无效的音频文件"})
            return result
        
        # 检查模型是否加载
        if not self.model:
            result.update({"status": "error", "error": "模型未加载"})
            return result
        
        try:
            self.logger.info(f"开始转写文件: {task.audio_file} (任务ID: {task.task_id})")
            start_time = time.time()
            
            # 执行语音转写
            transcription = self.model.transcribe(
                task.audio_file,
                language=self.config.language,
                fp16=(self.config.compute_type == "float16"),
                temperature=self.config.temperature
            )
            
            # 提取结果
            text = transcription.get("text", "").strip()
            duration = time.time() - start_time
            
            self.logger.info(
                f"转写完成，耗时{duration:.1f}s，字数{len(text)}，"
                f"置信度: {transcription.get('avg_logprob', 0):.2f}"
            )
            
            result.update({
                "status": "success",
                "text": text,
                "duration": duration,
                "language": transcription.get("language", ""),
                "confidence": transcription.get("avg_logprob", 0),
                "segments": [
                    {
                        "start": seg["start"],
                        "end": seg["end"],
                        "text": seg["text"].strip()
                    } for seg in transcription.get("segments", [])
                ]
            })
            
        except whisper.WhisperException as we:
            self.logger.error(f"Whisper转写错误: {we}", exc_info=True)
            result.update({"status": "error", "error": f"Whisper错误: {str(we)}"})
        except RuntimeError as re:
            if "CUDA out of memory" in str(re):
                self.logger.error("GPU内存不足，尝试清理缓存")
                torch.cuda.empty_cache()
                gc.collect()
                result.update({"status": "retry", "error": "GPU内存不足"})
            else:
                self.logger.error(f"运行时错误: {re}", exc_info=True)
                result.update({"status": "error", "error": f"运行时错误: {str(re)}"})
        except Exception as e:
            self.logger.error(f"转写失败: {e}", exc_info=True)
            result.update({"status": "error", "error": str(e)})
        finally:
            # 清理音频文件
            try:
                if os.path.exists(task.audio_file):
                    os.remove(task.audio_file)
                    self.logger.debug(f"已删除音频文件: {task.audio_file}")
            except Exception as e:
                self.logger.warning(f"删除音频文件失败: {e}")
        
        return result

    def run(self):
        """主运行循环"""
        self.logger.info("STT模块启动，等待转写任务...")
        
        while not self.stop_event.is_set():
            try:
                # 检查系统资源
                if not self.resource_monitor.check_resources():
                    time.sleep(1)
                    continue
                
                # 获取任务
                task: TranscriptionTask = self.command_queue.get(timeout=1)
                
                # 验证任务类型
                if not isinstance(task, TranscriptionTask):
                    self.logger.warning(f"无效任务类型: {type(task)}")
                    continue
                
                # 处理任务
                with self.lock:
                    if task.task_id in self.active_tasks:
                        self.logger.info(f"任务 {task.task_id} 已在处理中，跳过重复任务")
                        continue
                    
                    self.active_tasks[task.task_id] = task
                
                # 执行转写
                result = self.transcribe_audio(task)
                
                # 发送结果
                if task.callback_queue:
                    task.callback_queue.put(result)
                else:
                    self.result_queue.put(result)
                
                # 清理任务记录
                with self.lock:
                    if task.task_id in self.active_tasks:
                        del self.active_tasks[task.task_id]
                
                # 定期清理缓存
                if len(self.active_tasks) % 10 == 0:
                    torch.cuda.empty_cache()
                    gc.collect()

            except queue.Empty:
                # 定期健康检查
                if time.time() % self.config.health_check_interval < 1:
                    self.logger.info("健康检查: 模块运行正常")
                continue
            except Exception as e:
                self.logger.error(f"STT处理异常: {e}", exc_info=True)
        
        self.logger.info("STT模块停止运行")

def stt_process(command_queue: mp.Queue, result_queue: mp.Queue, 
               stop_event: mp.Event, config_dict: Dict = None):
    """STT进程入口函数"""
    # 设置进程名称
    mp.current_process().name = "STTProcess"
    
    # 加载配置
    config = STTConfig()
    if config_dict:
        config = STTConfig.from_dict(config_dict)
    
    # 初始化日志
    logger = setup_logger(
        Path(config.log_file) if config.log_file else None,
        config.log_level
    )
    
    # 优雅退出处理
    def signal_handler(sig, frame):
        logger.info("接收到终止信号，准备退出...")
        stop_event.set()
    
    signal.signal(signal.SIGINT, signal_handler)
    signal.signal(signal.SIGTERM, signal_handler)
    
    try:
        # 创建并运行STT模块
        stt = STTModule(command_queue, result_queue, stop_event, config)
        stt.run()
    except Exception as e:
        logger.exception(f"STT进程发生致命错误: {e}")
    finally:
        # 清理资源
        if hasattr(stt, 'model') and stt.model:
            del stt.model
            torch.cuda.empty_cache()
        logger.info("STT进程退出")
